Smart Routing Logic

Student Request ↓ [aiProviderRouter.js] ├── Lesson generation → Google Gemini (large context) ├── Chat/conversation → Groq (ultra-low latency) ├── Quiz generation → OpenRouter (structured output) ├── Grammar checking → SambaNova (reasoning) ├── Answer evaluation → Cerebras (high volume) └── All fallback → Next available provider

Failover Logic

async function getAIResponse(taskType, prompt) {
  const providers = getPriorityOrder(taskType);
  for (const provider of providers) {
    try {
      if (await hasRemainingQuota(provider)) {
        return await provider.call(prompt);
      }
    } catch (error) {
      logFailure(provider, error);
      continue;
    }
  }
  return getFallbackResponse(taskType, prompt);
}

class LessonAIService {
  async generateLesson({ topic, level, skillArea, duration }) {
    const prompt = this.promptBuilder.build('lessonGeneration', {
      topic, level, skillArea, duration,
      numVocabulary: 10, numGrammarPoints: 3
    });
    const examples = await this.getFewShotExamples('lessons', level);
    prompt.addExamples(examples);
    return await this.providerManager.call('GEMINI', prompt);
  }
}
